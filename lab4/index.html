import {createRouter, createWebHistory} from 'vue-router'
import HomeView from '../views/HomeView.vue'

const routes = [
    {
        path: '/',
        name: 'home',
        component: HomeView
    },
    {
        path: '/profile',
        name: 'profile',
        component: () => import('@/views/ProfileView.vue')
    },
    {
        path: '/readers',
        name: 'readers',
        component: () => import('@/views/ReadersView.vue')
    },
    {
        path: '/books',
        name: 'books',
        component: () => import('@/views/BooksView.vue')
    },
    {
        path: '/instances',
        name: 'instances',
        component: () => import('@/views/InstancesView.vue')
    },
]

const router = createRouter({
    history: createWebHistory(process.env.BASE_URL),
    routes
})

export default router

<script setup>
import {onMounted, ref} from "vue";
import api from "@/api";
import services from "@/services";

const headers = ref([
  {title: 'ID', key: 'id_book'},
  {title: 'Name', key: 'name'},
  {title: 'Author', key: 'author'},
  {title: 'Publisher', key: 'publisher'},
  {title: 'Actions', key: 'actions', sortable: false},
]);

const items = ref([]);

const editedItem = ref({
  name: '',
  author: '',
  publisher: '',
});

const fetchData = async () => {
  try {
    const response = await api.books.get();
    items.value = response.data;
  } catch (error) {
    console.log('', error)
  }
};

onMounted(fetchData);

const dialog = ref(false);

const save = async () => {
  try {
    console.log(editedItem.value);
    if (editedItem.value.id_book) {
      await api.books.update(editedItem.value.id_book, editedItem.value);
    } else {
      await api.books.create(editedItem.value);
    }
    await fetchData();
    dialog.value = false;
  } catch (error) {
    snackbar.value.text = `Failed! ${services.handeError(error.response)}`;
    snackbar.value.state = true;
  }
};

const dialogDelete = ref(false);

const deleteItem = ref(-1);

const openDialogDelete = (item) => {
  deleteItem.value = item.id_book;
  dialogDelete.value = true;
}

const deleteItemConfirm = async () => {
  try {
    await api.books.delete(deleteItem.value);
    dialogDelete.value = true;
    deleteItem.value = -1;
    await fetchData();
    dialogDelete.value = false;

  } catch (error) {
    snackbar.value.text = `Failed! ${services.handeError(error.response)}`;
    snackbar.value.state = true;
  }
}

const editItem = (item) => {
  editedItem.value = Object.assign({}, item);
  dialog.value = true;
}

const closeDialog = () => {
  dialog.value = false;
  editedItem.value = {
    name: '',
    author: '',
    publisher: '',
  };
}

const snackbar = ref({
  text: '',
  state: false,
});
</script>

<script setup>
import {computed, ref} from 'vue';
import {useStore} from "vuex";
import services from "@/services";

const store = useStore();

const formData = ref({
  username: '',
  password: '',
  confirmPassword: '',
});

const snackbar = ref({
  text: '',
  state: false,
});

const dialog = computed({
  get: () => store.getters['dialog/registerDialogState'],
  set: (value) => {
    store.dispatch('dialog/registerDialogSetState', value);
  }
});

const register = async () => {
  if (formData.value.password !== formData.value.confirmPassword) {
    snackbar.value.text = 'Failed! Passwords do not match!';
    snackbar.value.state = true;
    return;
  }

  try {
    console.log(formData.value);
    await store.dispatch('auth/register', formData.value);
    await store.dispatch('auth/login', formData.value)
    snackbar.value.text = 'Success!';
    snackbar.value.state = true;
    dialog.value = false;
  } catch (error) {
    snackbar.value.text = `Failed! ${services.handeError(error.response)}`;
    snackbar.value.state = true;
  }
}

</script>

<template>
  <v-dialog v-model="dialog" max-width="600px">
    <v-card>
      <v-card-title class="text-h5">Sign Up</v-card-title>
      <v-card-text>
        <v-form>
          <v-text-field v-model="formData.username" label="Username" required></v-text-field>
          <v-text-field v-model="formData.password" label="Password" required type="password"></v-text-field>
          <v-text-field v-model="formData.confirmPassword" label="Confirm Password" required
                        type="password"></v-text-field>
        </v-form>
      </v-card-text>
      <v-card-actions>
        <v-spacer></v-spacer>
        <v-btn color="blue darken-1" @click="register">Submit</v-btn>
        <v-btn color="red darken-1" @click="dialog = false">Cancel</v-btn>
      </v-card-actions>
    </v-card>
  </v-dialog>
  <v-snackbar v-model="snackbar.state" :timeout="3000">
    {{ snackbar.text }}
    <template v-slot:actions>
      <v-btn color="red" @click="snackbar.state = false">Close</v-btn>
    </template>
  </v-snackbar>
</template>

<script setup>
import {onMounted, ref} from "vue";
import api from "@/api";

const formData = ref({
  email: '',
  username: '',
});

onMounted(async () => {
  try {
    const response = await api.users.getCurrentUserData();
    formData.value.username = response.data.username;
    formData.value.email = response.data.email;
    console.log(formData.value);

  } catch (error) {
    console.error('getCurrentUserData', error);
  }
});

const isEditing = ref(false);

const saveChanges = async () => {
  try {
    await api.users.updateCurrentUserData(formData.value);
    isEditing.value = false;
  } catch (error) {
    console.error('updateCurrentUserData', error);
  }
};


</script>

<template>
  <v-container>
    <v-row>
      <v-col cols="12" md="4">
        <v-card>
          <v-card-title class="bg-grey-lighten-3">My Profile</v-card-title>
          <v-card-text>
            <v-list>
              <v-list-item v-if="!isEditing">
                <v-icon>mdi-email</v-icon>
                {{ formData.email }}
              </v-list-item>
              <v-text-field v-else v-model="formData.email" label="Email" required></v-text-field>
              <v-list-item v-if="!isEditing">
                <v-icon>mdi-account</v-icon>
                {{ formData.username }}
              </v-list-item>
              <v-text-field v-else v-model="formData.username" label="Username" required></v-text-field>
            </v-list>
          </v-card-text>
          <v-card-actions>
            <v-spacer></v-spacer>
            <v-btn v-if="!isEditing" @click="isEditing = true">Изменить</v-btn>
            <template v-else>
              <v-btn color="red" @click="isEditing = false">Отменить</v-btn>
              <v-btn color="green" @click="saveChanges">Сохранить</v-btn>
            </template>
          </v-card-actions>
        </v-card>
      </v-col>
    </v-row>
  </v-container>
</template>
